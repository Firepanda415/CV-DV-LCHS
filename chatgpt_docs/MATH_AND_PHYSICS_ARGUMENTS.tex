\section*{1. PDE Target and Discrete Mapping}
The continuous model is the 1D heat equation with Dirichlet boundaries:
\begin{equation}
\partial_t u(x,t)=\alpha\,\partial_x^2 u(x,t), \qquad u(0,t)=u(L,t)=0.
\end{equation}
After finite-difference mapping to 4 interior points (2 qubits), the target becomes
\begin{equation}
u_{\mathrm{PDE}}(t)=e^{-\alpha t T}u(0),
\end{equation}
with
\begin{equation}
T=
\begin{bmatrix}
2 & -1 & 0 & 0\\
-1 & 2 & -1 & 0\\
0 & -1 & 2 & -1\\
0 & 0 & -1 & 2
\end{bmatrix}
=2(I\otimes I)-(I\otimes X)-\frac12(X\otimes X+Y\otimes Y).
\end{equation}
Project objective: recover the PDE solution vector above, not a modified operator.

\section*{2. CV Preparation and Kernelized Coefficients}
The non-Gaussian CV preparation is expressed in the squeezed Fock basis:
\begin{equation}
\ket{\psi_{\mathrm{CV}}}=\sum_{n=0}^{N-1} C_n \ket{n},
\end{equation}
with coefficients generated by a kernel-weighted Hermite projection:
\begin{equation}
C_n \propto \int dk\;
e^{-\gamma k^2}\,g_{\beta_k}(k)\,
\frac{H_n(k/e^{r'})}{\sqrt{2^n n!}},
\quad
\gamma=e^{-2r'}-e^{-2r},\quad r'<r.
\end{equation}
Kernel:
\begin{equation}
g_{\beta_k}(k)=\frac{\exp\!\left((1+ik)^{\beta_k}\right)}{1-ik}.
\end{equation}
Interpretation:
\begin{itemize}
\item $r=r_{\mathrm{target}}$: post-selection squeezing scale.
\item $r'=r_{\mathrm{prime}}$: prep-basis squeezing scale.
\item $\beta_k=\texttt{kernel\_beta}$: spectral shaping of coefficient distribution.
\end{itemize}

\section*{3. Mixed-State Postselection Model}
Backend-feasible mode uses incoherent Fock mixture approximation:
\begin{equation}
\rho_{\mathrm{in}}\approx\sum_{n=0}^{N-1}|C_n|^2\ket{n}\bra{n}.
\end{equation}
Given joint unitary $U$ and CV projector $\Pi_{\phi}=\ket{\phi}\bra{\phi}$, the conditional DV state is
\begin{equation}
\rho_{\mathrm{post}}
=\frac{1}{p_{\mathrm{post}}}
\Tr_{\mathrm{CV}}\!\left[
(\Pi_\phi\otimes I)\,U(\rho_{\mathrm{in}}\otimes \ket{b}\bra{b})U^\dagger
\right],
\end{equation}
\begin{equation}
p_{\mathrm{post}}=
\Tr\!\left[
(\Pi_\phi\otimes I)\,U(\rho_{\mathrm{in}}\otimes \ket{b}\bra{b})U^\dagger
\right].
\end{equation}
Because $\rho_{\mathrm{in}}$ is mixed here, $\rho_{\mathrm{post}}$ is generally mixed.

\section*{4. Correctness Metrics}
Primary metric (absolute PDE-vector matching):
\begin{equation}
\varepsilon_{\mathrm{PDE}}
=\frac{\norm{u_{\mathrm{PDE}}-\sqrt{p_{\mathrm{post}}}\,\psi_{\mathrm{principal}}}}
{\norm{u_{\mathrm{PDE}}}},
\end{equation}
where $\psi_{\mathrm{principal}}$ is the principal eigenvector of $\rho_{\mathrm{post}}$.

Mixed-state directional diagnostic:
\begin{equation}
F_{\mathrm{mix}}
=\mel{\hat u_{\mathrm{PDE}}}{\rho_{\mathrm{post}}}{\hat u_{\mathrm{PDE}}},
\qquad
\hat u_{\mathrm{PDE}}=\frac{u_{\mathrm{PDE}}}{\norm{u_{\mathrm{PDE}}}}.
\end{equation}
Purity diagnostic:
\begin{equation}
\mathcal{P}=\Tr(\rho_{\mathrm{post}}^2).
\end{equation}
Project policy: optimize $\varepsilon_{\mathrm{PDE}}$ first, use $p_{\mathrm{post}}$ as secondary criterion.

\section*{5. Physics-Constrained Search Region}
Hard consistency:
\begin{equation}
r' < r
\quad\Longleftrightarrow\quad
\gamma(r,r')=e^{-2r'}-e^{-2r}>0.
\end{equation}
To avoid near-degenerate unstable shaping:
\begin{equation}
\gamma(r,r')\ge \gamma_{\min}.
\end{equation}

Effective support proxy:
\begin{equation}
n_{\mathrm{eff}}=\frac{1}{\sum_n |C_n|^4},
\qquad
n_{\mathrm{eff}}\le n_{\mathrm{eff,max}},
\end{equation}
which limits practical mode spread relative to finite cutoff $N=32$.

Postselection feasibility:
\begin{equation}
p_{\mathrm{post}}\ge p_{\min}.
\end{equation}

\section*{6. Optimization Model Implemented in Code}
Let $\theta=(r,r',\beta_k)$. Implemented penalty objective:
\begin{equation}
\begin{aligned}
J(\theta)=\varepsilon_{\mathrm{PDE}}(\theta)
&+\lambda_p\frac{[p_{\min}-p_{\mathrm{post}}(\theta)]_+}{p_{\min}}
+\lambda_{\gamma}\frac{[\gamma_{\min}-\gamma(\theta)]_+}{\gamma_{\min}}\\
&+\lambda_n\frac{[n_{\mathrm{eff}}(\theta)-n_{\mathrm{eff,max}}]_+}{n_{\mathrm{eff,max}}}
-10^{-3}p_{\mathrm{post}}(\theta),
\end{aligned}
\end{equation}
with $[x]_+=\max(0,x)$.

Equivalent constrained statement:
\begin{equation}
\min_{\theta}\ \varepsilon_{\mathrm{PDE}}(\theta)
\quad\text{s.t.}\quad
p_{\mathrm{post}}\ge p_{\min},\
\gamma\ge\gamma_{\min},\
n_{\mathrm{eff}}\le n_{\mathrm{eff,max}}.
\end{equation}

\section*{7. Commutator Structure and Trotter Scale}
Write
\begin{equation}
T = H_I + H_{IX} + H_{XX} + H_{YY},
\end{equation}
with
\begin{equation}
H_I=2(I\otimes I),\quad
H_{IX}=-(I\otimes X),\quad
H_{XX}=-\frac12(X\otimes X),\quad
H_{YY}=-\frac12(Y\otimes Y).
\end{equation}

The only non-zero pairwise commutator is
\begin{equation}
[H_{IX},H_{YY}]
=\left(-1\right)\left(-\frac12\right)[I\otimes X,\;Y\otimes Y]
=\frac12\left(2i\,Y\otimes Z\right)
= i\,Y\otimes Z,
\end{equation}
so
\begin{equation}
\norm{[H_{IX},H_{YY}]}=1.
\end{equation}
All other commutators vanish, hence
\begin{equation}
C_{\mathrm{comm}}=\sum_{j<\ell}\norm{[H_j,H_\ell]}=1.
\end{equation}

First-order Lie--Trotter operator error therefore scales as
\begin{equation}
\norm{e^{-\alpha t T}-\prod_j e^{-\alpha t H_j/n}}
\le
\frac{(\alpha t)^2}{2n}C_{\mathrm{comm}}
=\frac{(\alpha t)^2}{2n}.
\end{equation}
At $\alpha=t=1$, $n=100$, this gives $\le 0.005$ at operator level.

\section*{8. Practical Computational Strategy}
Due to expensive circuit evaluations, optimization is staged:
\begin{enumerate}
\item Global Latin-hypercube exploration over $(r,r',\beta_k)$.
\item Multi-start local Nelder--Mead refinement from best global seeds.
\item PDE-priority ranking and Pareto-front extraction in $(\varepsilon_{\mathrm{PDE}},p_{\mathrm{post}})$.
\end{enumerate}
This is a tractable approximation to global constrained optimization under costly black-box physics evaluations.

\paragraph{Important caveat.}
For mixture/coherence analysis, coefficient-only quantities such as $n_{\mathrm{eff}}$ are
used as practical diagnostics, not strict trace-distance bounds on
$\rho_{\mathrm{ideal}}-\rho_{\mathrm{mixture}}$ without additional assumptions on sector overlaps/phases.

\section*{9. Rigor Classification and Runtime Model}
\paragraph{Strict bound vs proxy.}
We separate claims into:
\begin{itemize}
\item \textbf{Strict bounds}: inequalities proved from operator identities and stated assumptions
(e.g., Trotter bound with explicit $C_{\mathrm{comm}}$).
\item \textbf{Proxies}: informative diagnostics not guaranteed to bound state-distance errors
(e.g., $n_{\mathrm{eff}}$, coefficient-only coherence indicators).
\end{itemize}

\paragraph{Coefficient-only coherence proxy.}
Define
\begin{equation}
\chi_C \equiv \sum_{m\neq n}\abs{C_m C_n}
=\left(\sum_n \abs{C_n}\right)^2-1.
\end{equation}
This is useful for ranking likely mixture sensitivity, but is not itself a strict
bound on $\norm{\rho_{\mathrm{ideal}}-\rho_{\mathrm{mixture}}}_1$ without further
assumptions on sector overlaps/phases.

\paragraph{Workload scaling model.}
For wall-time planning, use
\begin{equation}
N_{\mathrm{eval}}
\approx
N_{\mathrm{global}}
 + N_{\mathrm{starts}}\cdot N_{\mathrm{local,avg}},
\end{equation}
\begin{equation}
t_{\mathrm{total}} \approx N_{\mathrm{eval}}\,t_{\mathrm{eval}},
\end{equation}
where $t_{\mathrm{eval}}$ is the measured mean cost of one full
$\theta=(r,r',\beta_k)$ evaluation.
This model is coarse but sufficient to size quick/standard/full optimization profiles.

\paragraph{Reproducibility statement.}
Every reported optimum should be accompanied by:
\begin{equation}
\mathcal{M}=\{\text{commit},\ \text{environment},\ \text{full command},\ \text{output directory},\ \text{invariant settings}\},
\end{equation}
so numerical claims are traceable and repeatable.

\section*{10. Overnight Run Diagnostics and Robust Optimization Update}
\paragraph{Observed failure mode.}
In a long systematic run, local refinement terminated by backend synthesis error
(\texttt{HighLevelSynthesis} could not synthesize \texttt{cD} for one sampled point).
Hence, runtime was dominated by expensive local evaluations plus occasional backend-invalid points.

\paragraph{Robust objective handling.}
Define a backend-robust objective:
\begin{equation}
J_{\mathrm{robust}}(\theta)=
\begin{cases}
J(\theta), & \text{if circuit evaluation succeeds and outputs finite metrics},\\
\infty, & \text{otherwise}.
\end{cases}
\end{equation}
This keeps optimization mathematically consistent with constrained minimization while preventing a single backend fault from aborting the entire run.

\paragraph{Local-budget control and start diversity.}
To bound wall-time, impose
\begin{equation}
N_{\mathrm{eval,local}} \le N_{\mathrm{maxfev}}.
\end{equation}
To reduce redundant local starts in the same basin, choose starts with normalized-distance constraint
\begin{equation}
\norm{\tilde\theta_i-\tilde\theta_j}_2 \ge d_{\min},
\end{equation}
where $\tilde\theta$ is parameter-wise normalized to search bounds.

\section*{11. Fock-Cutoff Adequacy at the Current Best Basin}
For $\theta^\star=(r,r',\beta_k)\approx(0.46764,\,0.03022,\,0.49410)$, define truncation tail mass
\begin{equation}
M_{\mathrm{tail}}(N)=\sum_{n=N}^{\infty}\abs{C_n(\theta^\star)}^2.
\end{equation}
Numerically (using a higher-$N$ coefficient reference),
\begin{equation}
\sum_{n=0}^{31}\abs{C_n}^2 = 0.9999973462,\qquad
M_{\mathrm{tail}}(32)\approx 2.6538\times 10^{-6}.
\end{equation}
Hence the neglected coefficient mass beyond $N=32$ is very small in this basin.

For any bounded observable block contribution with $\norm{O_n}\le 1$, truncation-induced expectation error is bounded by
\begin{equation}
\abs{\Delta\!\expval{O}} \le M_{\mathrm{tail}}(N),
\end{equation}
so the coefficient-level truncation error scale at $N=32$ is in the few-$10^{-6}$ range here.

\paragraph{Practical implication.}
Use $N=32$ for optimization throughput; validate final top candidates at larger cutoff only as a final numerical sanity check.

\section*{12. Scope and Non-Overclaim Rules}
\paragraph{Basin scope.}
Numerical statements such as
\begin{equation}
\text{``small }r' \text{ basin gives } \eps_{\mathrm{PDE}}\approx 0.069\text{''}
\end{equation}
are empirical observations for the current ansatz/backend/mixture path and tested ranges.
They are not theorem-level lower bounds for all CV-DV LCHS variants.

\paragraph{Truncation scope.}
Statements like ``$N=16$ is enough'' must be tagged as basin-specific.
Project-wide reproducibility still uses fixed invariant $N=32$ unless a dedicated truncation study is being run.

\paragraph{Optimizer-domain consistency.}
If empirical best points satisfy, e.g., $r' \approx 0.03$, then optimizer lower bounds must include that regime.
Otherwise, constrained optimization can be internally correct but externally miss the best known basin.
